  <div id="squares-container">

    <!-- The 6 squares will be generated here -->

  </div>

  <div id="controls">

    <button id="flip-button">Flip Squares</button>

    <button id="reset-button" disabled>Reset</button>

  <p id="sample-list"></p>

  <p id="result"></p>

  

  <script>

    // Original data values

    const dataValues = [18, 24, 4, 30, 8, 42];

    const squaresContainer = document.getElementById('squares-container');

    const flipButton = document.getElementById('flip-button');

    const resetButton = document.getElementById('reset-button');

    const instructionP = document.getElementById('instruction');

    const sampleCountP = document.getElementById('sample-count');

    const sampleListP = document.getElementById('sample-list');

    const resultP = document.getElementById('result');


    let bootstrapSample = [];

    let selections = 0;


    // Function to create the squares

    function createSquares() {

      squaresContainer.innerHTML = '';

      dataValues.forEach((value) => {

        const square = document.createElement('div');

        square.classList.add('square');

        // Initially display the value

        square.textContent = value;

        square.dataset.value = value;

        // Add click event listener for sampling with replacement

        square.addEventListener('click', () => {

          if (square.classList.contains('hidden') && selections < 6) {

            // Reveal the value

            square.style.color = '#000';

            square.textContent = square.dataset.value;

            // Record the selected sample value

            bootstrapSample.push(Number(square.dataset.value));

            selections++;

            sampleCountP.textContent = `Samples chosen:

            // After a short delay, hide the value and shuffle squares

            setTimeout(() => {

              square.textContent = '';

              // Only shuffle if there are still selections to be made

              if (selections < 6) {

                shuffleSquares();

              }

            }, 800);


            // When 6 selections are made, calculate and display the average

            if (selections === 6) {

              const sum = bootstrapSample.reduce((acc, curr) => acc + curr, 0);

              const average = sum / bootstrapSample.length;

              resultP.textContent = `Bootstrap Sample Average: ${average.toFixed(2)}`;

              instructionP.textContent = 'Sampling complete. Click Reset to try again.';

              resetButton.disabled = false;

            }

          }

        });

        squaresContainer.appendChild(square);

      });

    }


    // Function to shuffle the squares

    function shuffleSquares() {

      const squares = Array.from(squaresContainer.children);

      squares.sort(() => Math.random() - 0.5);

      squaresContainer.innerHTML = '';

      squares.forEach(square => {

        squaresContainer.appendChild(square);

      });

    }

    

    // Initialize squares on page load

    createSquares();

    

    // Flip button hides the values on all squares

    flipButton.addEventListener('click', () => {

      document.querySelectorAll('.square').forEach(square => {

        square.classList.add('hidden');

        square.textContent = ''; // hide the text to simulate a flip

      });

      instructionP.textContent = 'Now click on a square to select a sample. Note: The squares shuffle automatically after each selection.';

      flipButton.disabled = true;

    });

    

    // Reset button to restart the activity

    resetButton.addEventListener('click', () => {

      bootstrapSample = [];

      selections = 0;

      sampleCountP.textContent = '';

      sampleListP.textContent = '';

      resultP.textContent = '';

      instructionP.textContent = 'Click the "Flip Squares" button to start sampling.';

      flipButton.disabled = false;

      resetButton.disabled = true;

      createSquares();

    });

  </script>
